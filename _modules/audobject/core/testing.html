      name: test

      - 0
      - 0
      pi: 3.1416

    """
    @init_decorator(
        resolvers={
            'point': resolver.Tuple,
        },

            self,
            name: str,
            *,
            point: (int, int) = (0, 0),
            **kwargs,
    ):
        super().__init__(**kwargs)

        self.point = point
